// ============================================================ // 장수한우곱창 자동 발주 시스템 — Google Apps Script // 파일명: 코드.gs // ============================================================ const CONFIG = { SOLAPI_API_KEY : 'YOUR_API_KEY', SOLAPI_API_SECRET : 'YOUR_API_SECRET', SENDER_NUMBER : '01041216995', VENDOR_NUMBER : '01053226995', OWNER_NUMBER : '01041216995', WONDANG: { WEEKDAY_TARGET : 8, WEEKEND_TARGET : 13, }, // 공공데이터포털 API 키 — 발급 후 입력 HOLIDAY_API_KEY : '', SHEET_ORDER : '발주기록', SHEET_STOCK : '입고기록', }; // ── ① 브라우저 접속 시 HTML 반환 ──────────────────────── function doGet(e) { return HtmlService .createHtmlOutputFromFile('index') .setTitle('원당점 발주·입고') .setXFrameOptionsMode(HtmlService.XFrameOptionsMode.ALLOWALL); } // ── ② POST 수신 ────────────────────────────────────────── function doPost(e) { try { const data = JSON.parse(e.postData.contents); if (data.type === 'order') return handleOrder(data); if (data.type === 'stock') return handleStock(data); return jsonResponse({ ok: false, message: '알 수 없는 type' }); } catch (err) { Logger.log('doPost 오류: ' + err.message); return jsonResponse({ ok: false, message: err.message }); } } // ── ③ 발주 처리 ────────────────────────────────────────── function handleOrder(data) { const gcUse = parseFloat(data.gc_use) || 0; const gcDone = parseFloat(data.gc_done) || 0; const gcIng = parseFloat(data.gc_ing) || 0; const dc = parseFloat(data.dc) || 0; const mc = parseFloat(data.mc) || 0; const extras = data.extras || []; const now = new Date(); const dateStr = formatDate(now); const dayOfWeek = now.getDay(); const currentStock = gcUse + gcDone + gcIng; // ── 공휴일 보정 포함 목표재고 계산 ── const target = getTargetWithHoliday(now, dayOfWeek, CONFIG.WONDANG); const rawOrder = target - currentStock; const calcOrder = rawOrder <= 0 ? 0 : Math.round(rawOrder); // 최소 2보 주문 (재고가 목표치 이상이어도 무조건 최소 2보) const gcOrder = Math.max(calcOrder, 2); logOrderToSheet(dateStr, gcUse, gcDone, gcIng, currentStock, target, gcOrder, dc, mc, extras); sendSms(CONFIG.VENDOR_NUMBER, buildVendorMsg(dateStr, gcOrder, dc, mc, extras)); sendSms(CONFIG.OWNER_NUMBER, buildOwnerOrderMsg(dateStr, gcUse, gcDone, gcIng, currentStock, target, gcOrder, dc, mc, extras)); return jsonResponse({ ok: true, gc_order: gcOrder, target: target }); } // ── ④ 입고 처리 ────────────────────────────────────────── function handleStock(data) { const gc = parseInt(data.gc) || 0; const dc = parseInt(data.dc) || 0; const mc = parseInt(data.mc) || 0; const extras = data.extras || []; const dateStr = formatDate(new Date()); logStockToSheet(dateStr, gc, dc, mc, extras); sendSms(CONFIG.OWNER_NUMBER, buildOwnerStockMsg(dateStr, gc, dc, mc, extras)); return jsonResponse({ ok: true }); } // ══════════════════════════════════════════════════════════ // ⑤ 공휴일 보정 목표재고 계산 // ══════════════════════════════════════════════════════════ function getTargetWithHoliday(today, dayOfWeek, branchConfig) { // 기본 목표 const base = (dayOfWeek === 5 || dayOfWeek === 6) ? branchConfig.WEEKEND_TARGET : branchConfig.WEEKDAY_TARGET; // 내일부터 7일 내 공휴일/연휴 여부 체크 const hasHoliday = checkUpcomingHoliday(today); if (hasHoliday) { Logger.log('공휴일 감지 → 목표재고 +1보 보정'); return base + 1; } return base; } // 내일~7일 내 공휴일 또는 토/일 연속 여부 확인 function checkUpcomingHoliday(today) { const year = today.getFullYear(); const month = today.getMonth() + 1; // 공공데이터 API 키가 있으면 실제 공휴일 체크 if (CONFIG.HOLIDAY_API_KEY) { try { const holidays = getPublicHolidays(year, month); // 다음달도 체크 (월말 기준) const nextMonth = month === 12 ? 1 : month + 1; const nextYear = month === 12 ? year + 1 : year; const holidaysNext = getPublicHolidays(nextYear, nextMonth); const allHolidays = holidays.concat(holidaysNext); // 내일부터 7일 내 공휴일 있는지 확인 for (let i = 1; i <= 7; i++) { const checkDate = new Date(today); checkDate.setDate(today.getDate() + i); const yy = checkDate.getFullYear(); const mm = String(checkDate.getMonth() + 1).padStart(2, '0'); const dd = String(checkDate.getDate()).padStart(2, '0'); const dateKey = `${yy}${mm}${dd}`; if (allHolidays.includes(dateKey)) { Logger.log('공휴일 감지: ' + dateKey); return true; } } } catch (err) { Logger.log('공휴일 API 오류 (토/일 체크로 대체): ' + err.message); return checkWeekendComing(today); } } // API 키 없으면 토/일 연속 체크로 대체 return checkWeekendComing(today); } // API 키 없을 때 — 내일이 금요일이면 주말 연휴 예고로 간주 function checkWeekendComing(today) { const tomorrow = new Date(today); tomorrow.setDate(today.getDate() + 1); const dow = tomorrow.getDay(); // 내일이 금(5) 또는 토(6)이면 주말 연휴 예정 return (dow === 5 || dow === 6); } // 공공데이터포털 공휴일 API 호출 function getPublicHolidays(year, month) { const url = 'https://apis.data.go.kr/B090041/openapi/service/SpcdeInfoService/getRestDeInfo' + '?serviceKey=' + encodeURIComponent(CONFIG.HOLIDAY_API_KEY) + '&solYear=' + year + '&solMonth=' + String(month).padStart(2, '0') + '&_type=json' + '&numOfRows=50'; const res = UrlFetchApp.fetch(url, { muteHttpExceptions: true }); const json = JSON.parse(res.getContentText()); const items = json?.response?.body?.items?.item; if (!items) return []; // item이 배열이 아닐 수도 있음 (1개일 때 객체로 옴) const list = Array.isArray(items) ? items : [items]; return list.map(item => String(item.locdate)); } // ── ⑥ 알림 트리거 (22시 / 15시) ───────────────────────── function sendOrderReminder() { const dateStr = formatDate(new Date()); const webAppUrl = ScriptApp.getService().getUrl(); sendSms('원당관리자번호입력', // ← 원당 관리자 번호로 교체 `[장수한우곱창]\n📋 발주 입력 시간입니다\n${dateStr}\n──────────────────\n아래 링크에서 재고를 입력해주세요\n${webAppUrl}` ); } function sendStockReminder() { const dateStr = formatDate(new Date()); const webAppUrl = ScriptApp.getService().getUrl(); sendSms('원당관리자번호입력', // ← 원당 관리자 번호로 교체 `[장수한우곱창]\n📦 입고 확인 시간입니다\n${dateStr}\n──────────────────\n아래 링크에서 입고량을 입력해주세요\n${webAppUrl}` ); } // ── ⑦ 문자 내용 생성 ───────────────────────────────────── function buildVendorMsg(dateStr, gcOrder, dc, mc, extras) { let msg = `[장수한우곱창 발주]\n원당점 · ${dateStr}\n──────────────────\n`; msg += `곱창 ${pad(gcOrder, 3)} 보\n`; if (dc > 0) msg += `대창 ${pad(dc, 3)} 보\n`; if (mc > 0) msg += `막창 ${pad(mc, 3)} 보\n`; if (extras.length) msg += `추가 ${extras.join(' / ')}\n`; msg += `──────────────────\n내일 납품 부탁드립니다.`; return msg; } function buildOwnerOrderMsg(dateStr, gcUse, gcDone, gcIng, currentStock, target, gcOrder, dc, mc, extras) { const isHolidayBoost = target > CONFIG.WONDANG.WEEKEND_TARGET ? false : (target % 1 === 0 && target !== getBaseTarget(new Date().getDay(), CONFIG.WONDANG)); let msg = `[발주완료] 원당점\n${dateStr}\n──────────────────\n곱창 재고\n`; msg += ` 쓰는것 ${pad(gcUse, 5)} 보\n`; msg += ` 연육완료 ${pad(gcDone, 5)} 보\n`; msg += ` 연육중 ${pad(gcIng, 5)} 보\n`; msg += ` 합계 ${pad(currentStock.toFixed(1), 5)} 보\n`; msg += ` 목표 ${pad(target, 5)} 보`; if (target !== getBaseTarget(new Date().getDay(), CONFIG.WONDANG)) { msg += ` (+1 연휴보정)`; } msg += `\n──────────────────\n`; msg += `곱창 발주 ${pad(gcOrder, 3)} 보\n`; if (dc > 0) msg += `대창 발주 ${pad(dc, 3)} 보\n`; if (mc > 0) msg += `막창 발주 ${pad(mc, 3)} 보\n`; if (extras.length) msg += `추가 ${extras.join(' / ')}\n`; msg += `──────────────────\n업체 문자 발송완료 ✅`; return msg; } function buildOwnerStockMsg(dateStr, gc, dc, mc, extras) { let msg = `[입고확인] 원당점\n${dateStr}\n──────────────────\n`; msg += `곱창 ${pad(gc, 3)} 보\n`; if (dc > 0) msg += `대창 ${pad(dc, 3)} 보\n`; if (mc > 0) msg += `막창 ${pad(mc, 3)} 보\n`; if (extras.length) msg += `기타 ${extras.join(' / ')}\n`; msg += `──────────────────\n입고 확인완료 ✅`; return msg; } // ── ⑧ 솔라피 SMS ───────────────────────────────────────── function sendSms(to, text) { const date = new Date().toISOString(); const salt = Utilities.getUuid(); const signature = computeHmac(CONFIG.SOLAPI_API_SECRET, date + salt); const res = UrlFetchApp.fetch('https://api.solapi.com/messages/v4/send', { method : 'post', contentType : 'application/json', headers : { 'Authorization': `HMAC-SHA256 apiKey=${CONFIG.SOLAPI_API_KEY}, date=${date}, salt=${salt}, signature=${signature}` }, payload : JSON.stringify({ message: { to, from: CONFIG.SENDER_NUMBER, text } }), muteHttpExceptions: true, }); const result = JSON.parse(res.getContentText()); if (result.errorCode) { Logger.log('SMS 오류: ' + JSON.stringify(result)); } else { Logger.log('SMS 발송완료 → ' + to); } } function computeHmac(secret, data) { const hash = Utilities.computeHmacSha256Signature( Utilities.newBlob(data).getBytes(), Utilities.newBlob(secret).getBytes() ); return hash.map(b => ('0' + (b & 0xff).toString(16)).slice(-2)).join(''); } // ── ⑨ 스프레드시트 기록 ────────────────────────────────── function logOrderToSheet(dateStr, gcUse, gcDone, gcIng, total, target, gcOrder, dc, mc, extras) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(CONFIG.SHEET_ORDER) || ss.insertSheet(CONFIG.SHEET_ORDER); if (sheet.getLastRow() === 0) { sheet.appendRow(['날짜','지점','쓰는것','연육완료','연육중','현재고','목표','발주(곱창)','대창','막창','추가주문','연휴보정']); } const baseTarget = getBaseTarget(new Date().getDay(), CONFIG.WONDANG); const isBoost = target !== baseTarget ? 'O' : '-'; sheet.appendRow([dateStr,'원당점',gcUse,gcDone,gcIng,total,target,gcOrder,dc,mc,extras.join(', '),isBoost]); } function logStockToSheet(dateStr, gc, dc, mc, extras) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(CONFIG.SHEET_STOCK) || ss.insertSheet(CONFIG.SHEET_STOCK); if (sheet.getLastRow() === 0) { sheet.appendRow(['날짜','지점','곱창 입고','대창 입고','막창 입고','기타']); } sheet.appendRow([dateStr,'원당점',gc,dc,mc,extras.join(', ')]); } // ── ⑩ 유틸 ────────────────────────────────────────────── function getBaseTarget(dayOfWeek, cfg) { return (dayOfWeek === 5 || dayOfWeek === 6) ? cfg.WEEKEND_TARGET : cfg.WEEKDAY_TARGET; } function formatDate(d) { const days = ['일','월','화','수','목','금','토']; return String(d.getFullYear()).slice(2) + '.' + String(d.getMonth()+1).padStart(2,'0') + '.' + String(d.getDate()).padStart(2,'0') + '(' + days[d.getDay()] + ')'; } function pad(val, len) { let s = String(val); while (s.length < len) s = ' ' + s; return s; } function jsonResponse(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); }
